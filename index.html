Create Simple Synth


					  function createSynth(stateObject) {
					    const context = new AudioContext();
					    const osc = context.createOscillator();
					    osc.frequency.value = 220;
					    osc.type = "sine";
					    const gain = context.createGain();
					    osc.connect(gain);
					    gain.connect(context.destination);
					    gain.gain.value = 0;
					    
					    context.addEventListener("statechange", () => {
					  	  if (context.state === "running") {
					  	  	stateObject.isSynthStarted = true;
					  	  }
					    });
					    
					    return {
					  	  context,
					  	  osc,
					  	  gain,
					    };
					  }

					function handlePlay(synth, stateObject) {
					  const playButton = document.querySelector(".play-btn");
					  playButton.addEventListener("click", () => {
					    if (!stateObject.isSynthStarted
						  && synth.gain.gain.value === 0) {
					      synth.osc.start(0);
					    }
					    synth.gain.gain.setTargetAtTime(
						    1,
						    synth.context.currentTime,
						    0.001
					    );
					  });
					}

					  function handlePause(synth) {
					    const pauseButton = document.querySelector(".pause-btn");
					    pauseButton.addEventListener("click", () => {
					  	const curTime = synth.context.currentTime;
					        synth.gain.gain.setTargetAtTime(0, curTime, 0.01);
					        synth.osc.stop(1);
					        stateObject.isSynthStarted = false;
					    });
					  }
				   

					  function handleInput(synth) {
					    const slider = document.getElementById("range");
					    const hz = document.querySelector('.range-hz');
					    slider.addEventListener("input", () => {
					      synth.osc.frequency.value = slider.value;
					      hz.textContent = `${slider.value} hz`;
					    });
					  }

					async function getMidiDevices(synth, stateObject) {
					  const midi = await navigator.requestMIDIAccess();
					  const inputs = midi.inputs;
					  const outputs = midi.outputs;
					  
					  getInOut(inputs, "LoopBe Internal MIDI", stateObject);
					  getInOut(outputs, "LoopBe Internal MIDI", stateObject);
					}
					  
					function getInOut(object, name, stateObject) {
					  object.forEach((value, key) => {
					      if (value.name === name) {
					          stateObject[value.type] = object.get(key);
					      }
					  });
					}

					function getMidiData(synth, stateObject) {
					  const targetInput = stateObject.input;

					  targetInput.addEventListener("midimessage", (message) => {
					    const [isOn, note, velocity] = message.data;
					    const frequency = Math.floor(440 * 2 ** ((note - 69) / 12));
					    synth.osc.frequency.value = frequency;
					    if (isOn === 128) {
					      synth.gain.gain.setTargetAtTime(
						      0,
						      synth.context.currentTime,
						      0.01
						    );
					      return;
					    }
					    synth.gain.gain.setTargetAtTime(
					      velocity / 100,
					      synth.context.currentTime,
					      0.001
					    );
					  });
					}
